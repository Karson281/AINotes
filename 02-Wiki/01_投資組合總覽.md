##  組合概況

```dataviewjs
const folder = "02-Wiki/Stocks";
const pages = dv.pages(`"${folder}"`).where(p => p.type === 'stock-analysis');

const tickers = pages.map(p => p.ticker).distinct().length;
const latestDate = pages.length > 0 ? pages.sort(p => p.date, 'desc')[0].date.toISODate() : 'N/A';
const todayCount = pages.where(p => p.date.toISODate() === latestDate).length;

const ratings = {};
pages.where(p => p.date.toISODate() === latestDate).forEach(p => {
  ratings[p.rating] = (ratings[p.rating] || 0) + 1;
});

dv.container.innerHTML = `
<div class="stats-grid">
  <div class="stat-card"><div class="stat-value">${tickers}</div><div class="stat-label">追蹤股票</div></div>
  <div class="stat-card"><div class="stat-value">${todayCount}</div><div class="stat-label">今日分析</div></div>
  <div class="stat-card stat-buy"><div class="stat-value">${ratings['建倉買入']||0}</div><div class="stat-label">建倉買入</div></div>
  <div class="stat-card stat-buy"><div class="stat-value">${ratings['加倉買入']||0}</div><div class="stat-label">加倉買入</div></div>
  <div class="stat-card stat-watch"><div class="stat-value">${ratings['密切觀察']||0}</div><div class="stat-label">密切觀察</div></div>
  <div class="stat-card stat-wait"><div class="stat-value">${ratings['觀望']||0}</div><div class="stat-label">觀望</div></div>
</div>`;
```

## 📋 今日分析

```dataviewjs
const folder = "02-Wiki/Stocks";
const pages = dv.pages(`"${folder}"`).where(p => p.type === 'stock-analysis').sort(p => p.date, 'desc');

if (pages.length === 0) {
  dv.paragraph('*尚無分析數據*');
} else {
  const latestDateStr = pages[0].date.toISODate();
  const todayPages = pages.where(p => p.date.toISODate() === latestDateStr);

  const getClass = (rating) => {
    if (rating === "建倉買入" || rating === "加倉買入") return {card: "rating-buy", badge: "badge-buy"};
    if (rating === "密切觀察") return {card: "rating-watch", badge: "badge-watch"};
    if (rating === "減倉賣出" || rating === "清倉賣出") return {card: "rating-sell", badge: "badge-sell"};
    return {card: "rating-wait", badge: "badge-wait"};
  };

  dv.container.innerHTML = '<div class="cards-grid">' + todayPages.map(p => {
    const style = getClass(p.rating);
    return `<div class="rating-card ${style.card}">
      <div class="card-header"><strong>${p.ticker}</strong><span class="rating-badge ${style.badge}">${p.rating}</span></div>
      <div class="card-body"><div>狀態: ${p.status||'N/A'}</div><div>來源: ${p.source||'N/A'}</div><div>日期: ${p.date.toISODate()}</div></div>
    </div>`;
  }).join('') + '</div>';
}
```

## 🗄️ 股票總表（每隻最新評級）

```dataviewjs
const folder = "02-Wiki/Stocks";
const pages = dv.pages(`"${folder}"`).where(p => p.type === 'stock-analysis').sort(p => p.date, 'desc');

const latestByTicker = {};
pages.forEach(p => {
  if (!latestByTicker[p.ticker] || p.date > latestByTicker[p.ticker].date) {
    latestByTicker[p.ticker] = p;
  }
});

const stocks = Object.values(latestByTicker).sort((a, b) => {
  const order = {'建倉買入':0,'加倉買入':1,'密切觀察':2,'觀望':3,'減倉賣出':4,'清倉賣出':5};
  return (order[a.rating]??99) - (order[b.rating]??99);
});

const getClass = (rating) => {
  if (rating === "建倉買入" || rating === "加倉買入") return {card: "rating-buy", badge: "badge-buy"};
  if (rating === "密切觀察") return {card: "rating-watch", badge: "badge-watch"};
  if (rating === "減倉賣出" || rating === "清倉賣出") return {card: "rating-sell", badge: "badge-sell"};
  return {card: "rating-wait", badge: "badge-wait"};
};

dv.container.innerHTML = `<table class="stock-table"><thead><tr>
  <th>代碼</th><th>評級</th><th>狀態</th><th>來源</th><th>日期</th>
</tr></thead><tbody>${stocks.map(p => {
  const style = getClass(p.rating);
  return `<tr><td><strong>${p.ticker}</strong></td><td><span class="rating-badge ${style.badge}">${p.rating}</span></td><td>${p.status||'-'}</td><td>${p.source||'-'}</td><td>${p.date.toISODate()}</td></tr>`;
}).join('')}</tbody></table>`;
```
