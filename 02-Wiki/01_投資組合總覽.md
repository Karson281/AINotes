```dataviewjs
const pages = dv.pages('"02-Wiki/Stocks"')
    .where(p => p.type == "stock-analysis" && p.date == dv.current().date)
    .sort(p => p.rating);

const ratings = {
    "建倉買入": { emoji: "🟢", stocks: [] },
    "加倉買入": { emoji: "🟢", stocks: [] },
    "密切觀察": { emoji: "🟡", stocks: [] },
    "觀望":     { emoji: "🟡", stocks: [] },
    "減倉賣出": { emoji: "🔴", stocks: [] },
    "清倉賣出": { emoji: "🔴", stocks: [] }
};

for (let p of pages) {
    if (ratings[p.rating]) {
        ratings[p.rating].stocks.push(p.ticker);
    }
}

const maxCount = Math.max(1, ...Object.values(ratings).map(r => r.stocks.length));
const barLen = 10;

for (let [r, data] of Object.entries(ratings)) {
    const count = data.stocks.length;
    const bar = "█".repeat(Math.round(count / maxCount * barLen)).padEnd(barLen, "░");
    const tickers = count > 0 ? " — " + data.stocks.join(", ") : "";
    dv.paragraph(`${data.emoji} **${r}**: ${count} 隻 \`${bar}\`${tickers}`);
}
```

```dataviewjs
const ratings = ["建倉買入", "加倉買入", "密切觀察", "觀望", "減倉賣出", "清倉賣出"];
const emoji = { "建倉買入": "🟢", "加倉買入": "🟢", "密切觀察": "🟡", "觀望": "🟡", "減倉賣出": "🔴", "清倉賣出": "🔴" };
const suggestion = { "建倉買入": "✅ 買入機會", "加倉買入": "✅ 加倉機會", "密切觀察": "👀 等信號", "觀望": "⏸️ 暫時不動", "減倉賣出": "⚠️ 減持", "清倉賣出": "🚨 清倉" };

const sections = [
    { name: "🟢 建倉 / 加倉", ratings: ["建倉買入", "加倉買入"] },
    { name: "🟡 持有觀察",    ratings: ["密切觀察", "觀望"] },
    { name: "🔴 減倉 / 清倉", ratings: ["減倉賣出", "清倉賣出"] }
];

const pages = dv.pages('"02-Wiki/Stocks"')
    .where(p => p.type == "stock-analysis" && p.date == dv.current().date);

for (let sec of sections) {
    const group = pages.filter(p => sec.ratings.includes(p.rating)).sort(p => p.ticker);
    if (group.length === 0) continue;

    dv.header(3, `${sec.name}（${group.length} 隻）`);
    dv.table(
        ["代號", "名稱", "現價", "變幅", "RSI(14)", "MA20", "MA50", "建議"],
        group.map(p => [
            p.ticker,
            p.name,
            p.price,
            p.change_percent + "%",
            p.rsi14,
            p.ma20,
            p.ma50,
            suggestion[p.rating] || ""
        ])
    );
}
```
