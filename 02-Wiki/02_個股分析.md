##  最新分析

```dataviewjs
function getDate(p) { return p.date?.toISODate ? p.date.toISODate() : String(p.date||''); }
const ticker = "0005.HK";
const pages = dv.pages('"02-Wiki/Stocks"')
  .where(p => p.ticker === ticker && p.type === 'stock-analysis')
  .sort(p => p.date, 'desc');

if (pages.length > 0) {
  const p = pages[0];
  dv.container.innerHTML = `
    <div class="rating-card rating-buy">
      <div class="card-header"><strong>${p.ticker}</strong><span class="rating-badge badge-buy">${p.rating}</span></div>
      <div class="card-body">
        <div>日期: ${getDate(p)} | 狀態: ${p.status}</div>
        <div>評級: ${p.rating}</div>
      </div>
    </div>`;
} else {
  dv.container.innerHTML = '<p style="color: var(--text-muted);">尚無分析數據</p>';
}
```
